```
class EventEmitter {
  constructor() {
    this.listeners = {};
  }
  // 注册事件 
  on(name, func) {
    if (!this.listeners[name]) {
      this.listeners[name] = [];
    }
    this.listeners[name].push(func);
  }
  // 触发事件
  emit(name, ...args) {
    if (this.listeners[name]) {
      this.listeners[name].forEach(listener => listener(...args));
    }
  }
  // 销毁事件
  off(name, func) {
    if (this.listeners[name]) {
      const index = this.listeners[name].indexOf(func);
      if (index > -1) {
       this.listeners[name].splice(index, 1); 
      }
    }
  }
}
```
